      cmdSearch: function () {
         var username = this.ui.txtSearchUsername.val();
 
         if (!username || !username.length) {
            this.ui.txtSearchUsername.focus();

         username = username.trim();
 
         if (!this.isValidGitHubUsername(username)) {
